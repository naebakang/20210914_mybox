# pull
docker pull [image name]

# run
docker run [image name]
docker run --name [image name]

# monitor
docker images
docker ps
docker ps -a
docker logs
docker logs -f 

# stop
docker stop [container name]

# start
docker start [container name]

# delete
docker rm [container name]
docker rm --force [container name]
docker rmi [image name]

# port forwarding
docker run -p 80:80 [image name]

# into container
docker exec [container name] pwd
docker exec -it [container name] /bin/bash
docker exec -it [container name] /bin/sh
docker attach [CONTAINER ID]

# out container
exit

# connect local to container
docker run -p 8080:80 -v ~/Desktop/htdocs:/usr/local/apache2/htdocs/ httpd

# create new image
docker commit [container name] [REPOSITORY:TAG]
docker build -t [new image name ] [REPOSITORY:TAG]  <-- need 'docker file'

# up load image
docker push [USER NAME/REPOSITORY:TAG]
